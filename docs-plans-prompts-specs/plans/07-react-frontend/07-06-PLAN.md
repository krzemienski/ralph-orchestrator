<reasoning_guidance>
Audio player is a complex interactive component. Thoroughly analyze before implementing:

**Browser Audio API Considerations:**
- HTMLAudioElement vs Web Audio API trade-offs
- Seeking accuracy (currentTime can be imprecise)
- Buffering states and loading indicators
- Preloading strategies (none, metadata, auto)

**UX Design Decisions:**
- Progress bar: click-to-seek vs drag handle vs both?
- Time display format: elapsed/remaining/both?
- Chapter navigation: list view, markers on progress bar, or both?
- Keyboard shortcuts (space=play/pause, arrows=seek)

**State Synchronization Challenges:**
- Audio currentTime vs React state (avoid render loops)
- Seeking while playing vs paused
- Buffered ranges visualization
- Chapter boundaries and auto-advance

**Accessibility Requirements:**
- Screen reader announcements for play state
- Keyboard navigation for all controls
- Focus management during chapter switches
- Time announced in accessible format

**Consider Existing Patterns:**
- Study Spotify, Apple Music, Audible interfaces
- Proven library options: react-player, howler.js, wavesurfer.js
- Custom implementation trade-offs (control vs maintenance)

The implementation here uses native HTMLAudioElement for simplicity - 
consider wavesurfer.js for waveform visualization if needed.
</reasoning_guidance>

```tsx
import { useState, useEffect, useRef } from "react";
import { useParams, Link } from "react-router-dom";
import { Button } from "@/components/ui/button";
import { Slider } from "@/components/ui/slider";
import { 
  Play, Pause, SkipBack, SkipForward, Volume2, VolumeX, 
  Download, Share2, ArrowLeft, Loader2, List
} from "lucide-react";
import { api } from "@/lib/api";

interface Chapter {
  id: string;
  chapter_number: number;
  title: string;
  start_time_seconds: number;
  duration_seconds: number;
}

interface Story {
  id: string;
  title: string;
  repository_url: string;
  style: string;
  total_duration_seconds: number;
  chapters: Chapter[];
}

export function StoryPlayer() {
  const { storyId } = useParams<{ storyId: string }>();
  const [story, setStory] = useState<Story | null>(null);
  const [audioUrl, setAudioUrl] = useState<string | null>(null);
  const [isLoading, setIsLoading] = useState(true);
  const [isPlaying, setIsPlaying] = useState(false);
  const [currentTime, setCurrentTime] = useState(0);
  const [duration, setDuration] = useState(0);
  const [volume, setVolume] = useState(1);
  const [isMuted, setIsMuted] = useState(false);
  const [activeChapter, setActiveChapter] = useState(0);
  const [showChapters, setShowChapters] = useState(true);
  const audioRef = useRef<HTMLAudioElement>(null);

  useEffect(() => {
    loadStory();
  }, [storyId]);

  async function loadStory() {
    try {
      const [storyRes, audioRes] = await Promise.all([
        api.get(`/stories/${storyId}`),
        api.get(`/audio/stories/${storyId}/manifest`),
      ]);
      
      setStory(storyRes.data);
      setAudioUrl(audioRes.data.main_audio_url);
    } catch (error) {
      console.error("Failed to load story:", error);
    } finally {
      setIsLoading(false);
    }
  }

  useEffect(() => {
    const audio = audioRef.current;
    if (!audio) return;

    const updateTime = () => {
      setCurrentTime(audio.currentTime);
      
      // Update active chapter
      if (story?.chapters) {
        const chapter = story.chapters.findIndex((ch, i) => {
          const next = story.chapters[i + 1];
          return audio.currentTime >= ch.start_time_seconds && 
                 (!next || audio.currentTime < next.start_time_seconds);
        });
        if (chapter !== -1) setActiveChapter(chapter);
      }
    };

    audio.addEventListener("timeupdate", updateTime);
    audio.addEventListener("loadedmetadata", () => setDuration(audio.duration));
    audio.addEventListener("ended", () => setIsPlaying(false));

    return () => {
      audio.removeEventListener("timeupdate", updateTime);
    };
  }, [story]);

  function togglePlay() {
    if (!audioRef.current) return;
    
    if (isPlaying) {
      audioRef.current.pause();
    } else {
      audioRef.current.play();
    }
    setIsPlaying(!isPlaying);
  }

  function seek(time: number) {
    if (!audioRef.current) return;
    audioRef.current.currentTime = time;
    setCurrentTime(time);
  }

  function seekToChapter(index: number) {
    if (!story?.chapters[index]) return;
    seek(story.chapters[index].start_time_seconds);
    setActiveChapter(index);
    if (!isPlaying) togglePlay();
  }

  function skipBack() {
    seek(Math.max(0, currentTime - 15));
  }

  function skipForward() {
    seek(Math.min(duration, currentTime + 30));
  }

  function handleVolumeChange(value: number[]) {
    const vol = value[0];
    setVolume(vol);
    if (audioRef.current) audioRef.current.volume = vol;
    setIsMuted(vol === 0);
  }

  function toggleMute() {
    if (audioRef.current) {
      if (isMuted) {
        audioRef.current.volume = volume || 1;
        setIsMuted(false);
      } else {
        audioRef.current.volume = 0;
        setIsMuted(true);
      }
    }
  }

  function formatTime(seconds: number): string {
    const mins = Math.floor(seconds / 60);
    const secs = Math.floor(seconds % 60);
    return `${mins}:${secs.toString().padStart(2, "0")}`;
  }

  async function handleDownload() {
    window.open(`${import.meta.env.VITE_API_URL}/audio/stories/${storyId}?download=true`, "_blank");
  }

  if (isLoading) {
    return (
      <div className="min-h-screen bg-slate-900 flex items-center justify-center">
        <Loader2 className="h-8 w-8 animate-spin text-indigo-400" />
      </div>
    );
  }

  if (!story || !audioUrl) {
    return (
      <div className="min-h-screen bg-slate-900 flex items-center justify-center">
        <p className="text-slate-400">Story not found</p>
      </div>
    );
  }

  const repoName = story.repository_url.split("/").slice(-2).join("/");

  return (
    <div className="min-h-screen bg-slate-900">
      <audio ref={audioRef} src={audioUrl} preload="metadata" />

      {/* Header */}
      <header className="bg-slate-800 border-b border-slate-700 px-4 py-4">
        <div className="container mx-auto flex items-center gap-4">
          <Link to="/dashboard">
            <Button variant="ghost" size="sm" className="text-slate-400 hover:text-white">
              <ArrowLeft className="h-4 w-4" />
            </Button>
          </Link>
          <div className="flex-1">
            <h1 className="text-lg font-semibold text-white">{story.title}</h1>
            <p className="text-sm text-slate-400">{repoName}</p>
          </div>
          <div className="flex items-center gap-2">
            <Button variant="ghost" size="sm" className="text-slate-400 hover:text-white">
              <Share2 className="h-4 w-4" />
            </Button>
            <Button 
              variant="ghost" 
              size="sm" 
              className="text-slate-400 hover:text-white"
              onClick={handleDownload}
            >
              <Download className="h-4 w-4" />
            </Button>
          </div>
        </div>
      </header>

      <main className="container mx-auto px-4 py-8">
        <div className="grid lg:grid-cols-3 gap-8">
          {/* Player */}
          <div className="lg:col-span-2">
            <div className="bg-slate-800 rounded-xl p-6 border border-slate-700">
              {/* Progress */}
              <div className="mb-6">
                <Slider
                  value={[currentTime]}
                  max={duration || 100}
                  step={1}
                  onValueChange={(v) => seek(v[0])}
                  className="w-full"
                />
                <div className="flex justify-between text-sm text-slate-400 mt-2">
                  <span>{formatTime(currentTime)}</span>
                  <span>{formatTime(duration)}</span>
                </div>
              </div>

              {/* Controls */}
              <div className="flex items-center justify-center gap-4 mb-6">
                <Button 
                  variant="ghost" 
                  size="lg" 
                  className="text-slate-400 hover:text-white"
                  onClick={skipBack}
                >
                  <SkipBack className="h-6 w-6" />
                </Button>
                
                <Button
                  size="lg"
                  className="w-16 h-16 rounded-full bg-indigo-600 hover:bg-indigo-500"
                  onClick={togglePlay}
                >
                  {isPlaying ? (
                    <Pause className="h-8 w-8" />
                  ) : (
                    <Play className="h-8 w-8 ml-1" />
                  )}
                </Button>

                <Button 
                  variant="ghost" 
                  size="lg" 
                  className="text-slate-400 hover:text-white"
                  onClick={skipForward}
                >
                  <SkipForward className="h-6 w-6" />
                </Button>
              </div>

              {/* Volume */}
              <div className="flex items-center justify-center gap-3">
                <Button 
                  variant="ghost" 
                  size="sm"
                  onClick={toggleMute}
                  className="text-slate-400 hover:text-white"
                >
                  {isMuted ? <VolumeX className="h-5 w-5" /> : <Volume2 className="h-5 w-5" />}
                </Button>
                <Slider
                  value={[isMuted ? 0 : volume]}
                  max={1}
                  step={0.1}
                  onValueChange={handleVolumeChange}
                  className="w-32"
                />
              </div>
            </div>

            {/* Current Chapter */}
            {story.chapters[activeChapter] && (
              <div className="mt-6 bg-slate-800/50 rounded-xl p-6 border border-slate-700">
                <div className="text-indigo-400 text-sm font-medium mb-1">
                  Chapter {activeChapter + 1}
                </div>
                <h2 className="text-xl font-semibold text-white">
                  {story.chapters[activeChapter].title}
                </h2>
              </div>
            )}
          </div>

          {/* Chapter List */}
          <div className="lg:col-span-1">
            <div className="bg-slate-800 rounded-xl border border-slate-700 overflow-hidden">
              <div className="p-4 border-b border-slate-700 flex items-center justify-between">
                <h3 className="font-semibold text-white flex items-center gap-2">
                  <List className="h-4 w-4" />
                  Chapters
                </h3>
                <span className="text-sm text-slate-400">
                  {story.chapters.length} chapters
                </span>
              </div>
              <div className="max-h-96 overflow-y-auto">
                {story.chapters.map((chapter, index) => (
                  <button
                    key={chapter.id}
                    onClick={() => seekToChapter(index)}
                    className={`w-full p-4 text-left border-b border-slate-700 last:border-0 transition-colors ${
                      index === activeChapter 
                        ? "bg-indigo-500/10 border-l-2 border-l-indigo-500" 
                        : "hover:bg-slate-700/50"
                    }`}
                  >
                    <div className="flex items-start gap-3">
                      <span className={`text-sm font-mono ${
                        index === activeChapter ? "text-indigo-400" : "text-slate-500"
                      }`}>
                        {index + 1}
                      </span>
                      <div>
                        <div className={index === activeChapter ? "text-white" : "text-slate-300"}>
                          {chapter.title}
                        </div>
                        <div className="text-xs text-slate-500 mt-1">
                          {formatTime(chapter.start_time_seconds)} Â· {Math.round(chapter.duration_seconds / 60)} min
                        </div>
                      </div>
                    </div>
                  </button>
                ))}
              </div>
            </div>
          </div>
        </div>
      </main>
    </div>
  );
}
```

```typescript
import { useEffect, useCallback } from "react";

interface ShortcutActions {
  togglePlay?: () => void;
  skipBack?: () => void;
  skipForward?: () => void;
  nextChapter?: () => void;
  prevChapter?: () => void;
  toggleMute?: () => void;
}

export function useKeyboardShortcuts(actions: ShortcutActions) {
  const handleKeyDown = useCallback((e: KeyboardEvent) => {
    // Ignore if typing in input
    if (e.target instanceof HTMLInputElement || e.target instanceof HTMLTextAreaElement) {
      return;
    }

    switch (e.code) {
      case "Space":
        e.preventDefault();
        actions.togglePlay?.();
        break;
      case "ArrowLeft":
        e.preventDefault();
        if (e.shiftKey) {
          actions.prevChapter?.();
        } else {
          actions.skipBack?.();
        }
        break;
      case "ArrowRight":
        e.preventDefault();
        if (e.shiftKey) {
          actions.nextChapter?.();
        } else {
          actions.skipForward?.();
        }
        break;
      case "KeyM":
        actions.toggleMute?.();
        break;
    }
  }, [actions]);

  useEffect(() => {
    document.addEventListener("keydown", handleKeyDown);
    return () => document.removeEventListener("keydown", handleKeyDown);
  }, [handleKeyDown]);
}
```
