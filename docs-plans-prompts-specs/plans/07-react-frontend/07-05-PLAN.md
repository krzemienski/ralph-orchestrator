```tsx
import { useState, useEffect } from "react";
import { Link, useNavigate } from "react-router-dom";
import { Button } from "@/components/ui/button";
import { Plus, Headphones, Clock, CheckCircle, Loader2, AlertCircle, Trash2, Play } from "lucide-react";
import { api } from "@/lib/api";
import { useAuth } from "@/hooks/useAuth";
import { formatDistanceToNow } from "date-fns";

interface Story {
  id: string;
  title: string;
  repository_url: string;
  status: "pending" | "analyzing" | "generating" | "synthesizing" | "complete" | "failed";
  style: string;
  created_at: string;
  completed_at: string | null;
  total_duration_seconds: number | null;
}

export function Dashboard() {
  const [stories, setStories] = useState<Story[]>([]);
  const [isLoading, setIsLoading] = useState(true);
  const { user, logout } = useAuth();
  const navigate = useNavigate();

  useEffect(() => {
    loadStories();
  }, []);

  async function loadStories() {
    try {
      const response = await api.get("/stories");
      setStories(response.data.stories);
    } catch (error) {
      console.error("Failed to load stories:", error);
    } finally {
      setIsLoading(false);
    }
  }

  async function deleteStory(storyId: string) {
    if (!confirm("Are you sure you want to delete this story?")) return;
    
    try {
      await api.delete(`/stories/${storyId}`);
      setStories((prev) => prev.filter((s) => s.id !== storyId));
    } catch (error) {
      console.error("Failed to delete story:", error);
    }
  }

  const inProgressStories = stories.filter((s) => 
    ["pending", "analyzing", "generating", "synthesizing"].includes(s.status)
  );
  const completedStories = stories.filter((s) => s.status === "complete");
  const failedStories = stories.filter((s) => s.status === "failed");

  return (
    <div className="min-h-screen bg-slate-900">
      {/* Header */}
      <header className="bg-slate-800 border-b border-slate-700">
        <div className="container mx-auto px-4 py-4 flex items-center justify-between">
          <div className="flex items-center gap-2">
            <Headphones className="h-6 w-6 text-indigo-400" />
            <span className="text-xl font-bold text-white">Code Story</span>
          </div>
          <div className="flex items-center gap-4">
            <span className="text-slate-400">{user?.name}</span>
            <Button variant="ghost" onClick={logout} className="text-slate-400 hover:text-white">
              Sign Out
            </Button>
          </div>
        </div>
      </header>

      <main className="container mx-auto px-4 py-8">
        {/* Actions */}
        <div className="flex items-center justify-between mb-8">
          <h1 className="text-2xl font-bold text-white">Your Stories</h1>
          <Link to="/new">
            <Button className="bg-indigo-600 hover:bg-indigo-500">
              <Plus className="h-4 w-4 mr-2" />
              New Story
            </Button>
          </Link>
        </div>

        {isLoading ? (
          <div className="flex items-center justify-center py-20">
            <Loader2 className="h-8 w-8 animate-spin text-indigo-400" />
          </div>
        ) : stories.length === 0 ? (
          <EmptyState />
        ) : (
          <div className="space-y-8">
            {/* In Progress */}
            {inProgressStories.length > 0 && (
              <section>
                <h2 className="text-lg font-semibold text-white mb-4 flex items-center gap-2">
                  <Loader2 className="h-5 w-5 animate-spin text-indigo-400" />
                  In Progress ({inProgressStories.length})
                </h2>
                <div className="grid gap-4">
                  {inProgressStories.map((story) => (
                    <StoryCard 
                      key={story.id} 
                      story={story} 
                      onDelete={deleteStory}
                    />
                  ))}
                </div>
              </section>
            )}

            {/* Completed */}
            {completedStories.length > 0 && (
              <section>
                <h2 className="text-lg font-semibold text-white mb-4 flex items-center gap-2">
                  <CheckCircle className="h-5 w-5 text-green-400" />
                  Completed ({completedStories.length})
                </h2>
                <div className="grid gap-4">
                  {completedStories.map((story) => (
                    <StoryCard 
                      key={story.id} 
                      story={story}
                      onDelete={deleteStory}
                    />
                  ))}
                </div>
              </section>
            )}

            {/* Failed */}
            {failedStories.length > 0 && (
              <section>
                <h2 className="text-lg font-semibold text-white mb-4 flex items-center gap-2">
                  <AlertCircle className="h-5 w-5 text-red-400" />
                  Failed ({failedStories.length})
                </h2>
                <div className="grid gap-4">
                  {failedStories.map((story) => (
                    <StoryCard 
                      key={story.id} 
                      story={story}
                      onDelete={deleteStory}
                    />
                  ))}
                </div>
              </section>
            )}
          </div>
        )}
      </main>
    </div>
  );
}

function StoryCard({ story, onDelete }: { story: Story; onDelete: (id: string) => void }) {
  const navigate = useNavigate();
  const repoName = story.repository_url.split("/").slice(-2).join("/");

  function handleClick() {
    if (story.status === "complete") {
      navigate(`/stories/${story.id}/player`);
    } else if (["pending", "analyzing", "generating", "synthesizing"].includes(story.status)) {
      navigate(`/stories/${story.id}/progress`);
    }
  }

  return (
    <div 
      className="bg-slate-800 rounded-xl p-5 border border-slate-700 hover:border-slate-600 cursor-pointer transition-all"
      onClick={handleClick}
    >
      <div className="flex items-start justify-between">
        <div className="flex-1">
          <h3 className="text-lg font-medium text-white mb-1">
            {story.title || repoName}
          </h3>
          <p className="text-slate-400 text-sm mb-3">{repoName}</p>
          
          <div className="flex items-center gap-4 text-sm">
            <StatusBadge status={story.status} />
            <span className="text-slate-500 capitalize">{story.style}</span>
            {story.total_duration_seconds && (
              <span className="text-slate-500 flex items-center gap-1">
                <Clock className="h-4 w-4" />
                {Math.round(story.total_duration_seconds / 60)} min
              </span>
            )}
            <span className="text-slate-500">
              {formatDistanceToNow(new Date(story.created_at), { addSuffix: true })}
            </span>
          </div>
        </div>

        <div className="flex items-center gap-2" onClick={(e) => e.stopPropagation()}>
          {story.status === "complete" && (
            <Button 
              size="sm" 
              variant="ghost" 
              className="text-indigo-400 hover:text-indigo-300"
              onClick={() => navigate(`/stories/${story.id}/player`)}
            >
              <Play className="h-4 w-4" />
            </Button>
          )}
          <Button 
            size="sm" 
            variant="ghost" 
            className="text-slate-400 hover:text-red-400"
            onClick={() => onDelete(story.id)}
          >
            <Trash2 className="h-4 w-4" />
          </Button>
        </div>
      </div>
    </div>
  );
}

function StatusBadge({ status }: { status: Story["status"] }) {
  const config = {
    pending: { color: "text-yellow-400 bg-yellow-400/10", label: "Pending" },
    analyzing: { color: "text-blue-400 bg-blue-400/10", label: "Analyzing" },
    generating: { color: "text-purple-400 bg-purple-400/10", label: "Generating" },
    synthesizing: { color: "text-indigo-400 bg-indigo-400/10", label: "Synthesizing" },
    complete: { color: "text-green-400 bg-green-400/10", label: "Complete" },
    failed: { color: "text-red-400 bg-red-400/10", label: "Failed" },
  };

  const { color, label } = config[status];

  return (
    <span className={`px-2 py-1 rounded-full text-xs font-medium ${color}`}>
      {label}
    </span>
  );
}

function EmptyState() {
  return (
    <div className="text-center py-20">
      <Headphones className="h-16 w-16 text-slate-600 mx-auto mb-4" />
      <h2 className="text-xl font-semibold text-white mb-2">No stories yet</h2>
      <p className="text-slate-400 mb-6">Create your first code story to get started</p>
      <Link to="/new">
        <Button className="bg-indigo-600 hover:bg-indigo-500">
          <Plus className="h-4 w-4 mr-2" />
          Create Your First Story
        </Button>
      </Link>
    </div>
  );
}
```

Create story progress page with WebSocket:

```tsx
import { useState, useEffect } from "react";
import { useParams, useNavigate } from "react-router-dom";
import { Loader2, CheckCircle, AlertCircle, Code, FileText, Mic, ArrowRight } from "lucide-react";
import { Button } from "@/components/ui/button";

interface ProgressData {
  status: string;
  progress_percent: number;
  current_step: string;
}

const STEPS = [
  { key: "analyzing", label: "Analyzing Repository", icon: Code },
  { key: "generating", label: "Generating Script", icon: FileText },
  { key: "synthesizing", label: "Creating Audio", icon: Mic },
];

export function StoryProgress() {
  const { storyId } = useParams<{ storyId: string }>();
  const [progress, setProgress] = useState<ProgressData>({
    status: "pending",
    progress_percent: 0,
    current_step: "Starting...",
  });
  const [error, setError] = useState<string | null>(null);
  const navigate = useNavigate();

  useEffect(() => {
    const token = localStorage.getItem("token");
    const wsUrl = `${import.meta.env.VITE_WS_URL}/ws/stories/${storyId}/progress?token=${token}`;
    
    const ws = new WebSocket(wsUrl);

    ws.onmessage = (event) => {
      const data = JSON.parse(event.data);
      setProgress(data);

      if (data.status === "complete") {
        setTimeout(() => navigate(`/stories/${storyId}/player`), 1500);
      } else if (data.status === "failed") {
        setError(data.error_message || "Story generation failed");
      }
    };

    ws.onerror = () => {
      setError("Connection lost. Please refresh the page.");
    };

    return () => ws.close();
  }, [storyId, navigate]);

  const currentStepIndex = STEPS.findIndex((s) => s.key === progress.status);

  return (
    <div className="min-h-screen bg-slate-900 flex items-center justify-center p-4">
      <div className="max-w-lg w-full">
        {error ? (
          <div className="text-center">
            <AlertCircle className="h-16 w-16 text-red-400 mx-auto mb-4" />
            <h1 className="text-2xl font-bold text-white mb-2">Generation Failed</h1>
            <p className="text-slate-400 mb-6">{error}</p>
            <Button onClick={() => navigate("/dashboard")} variant="outline">
              Back to Dashboard
            </Button>
          </div>
        ) : progress.status === "complete" ? (
          <div className="text-center">
            <CheckCircle className="h-16 w-16 text-green-400 mx-auto mb-4" />
            <h1 className="text-2xl font-bold text-white mb-2">Story Complete!</h1>
            <p className="text-slate-400 mb-6">Redirecting to player...</p>
          </div>
        ) : (
          <div className="bg-slate-800 rounded-xl p-8 border border-slate-700">
            <h1 className="text-2xl font-bold text-white mb-2 text-center">
              Creating Your Story
            </h1>
            <p className="text-slate-400 text-center mb-8">
              This usually takes 2-3 minutes
            </p>

            {/* Progress Bar */}
            <div className="mb-8">
              <div className="flex justify-between text-sm text-slate-400 mb-2">
                <span>{progress.current_step}</span>
                <span>{progress.progress_percent}%</span>
              </div>
              <div className="h-2 bg-slate-700 rounded-full overflow-hidden">
                <div 
                  className="h-full bg-indigo-500 transition-all duration-500"
                  style={{ width: `${progress.progress_percent}%` }}
                />
              </div>
            </div>

            {/* Steps */}
            <div className="space-y-4">
              {STEPS.map((step, index) => {
                const Icon = step.icon;
                const isActive = index === currentStepIndex;
                const isComplete = index < currentStepIndex;

                return (
                  <div 
                    key={step.key}
                    className={`flex items-center gap-4 p-4 rounded-lg ${
                      isActive ? "bg-indigo-500/10 border border-indigo-500/50" : "bg-slate-700/30"
                    }`}
                  >
                    <div className={`
                      w-10 h-10 rounded-full flex items-center justify-center
                      ${isComplete ? "bg-green-500" : isActive ? "bg-indigo-500" : "bg-slate-600"}
                    `}>
                      {isComplete ? (
                        <CheckCircle className="h-5 w-5 text-white" />
                      ) : isActive ? (
                        <Loader2 className="h-5 w-5 text-white animate-spin" />
                      ) : (
                        <Icon className="h-5 w-5 text-slate-400" />
                      )}
                    </div>
                    <span className={`font-medium ${
                      isActive ? "text-white" : isComplete ? "text-slate-300" : "text-slate-500"
                    }`}>
                      {step.label}
                    </span>
                  </div>
                );
              })}
            </div>
          </div>
        )}
      </div>
    </div>
  );
}
```
